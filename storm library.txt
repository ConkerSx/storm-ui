-- UI Library compacta e vertical para PC e Mobile
-- Suporta: Button, Toggle, Slider, Dropdown, MultiDropdown, Keybind, TextBox, ColorPicker, Label
-- Minimizar: clique no botão "-" para esconder/mostrar os elementos

local UserInputService = game:GetService("UserInputService")
local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer

local Library = {}

function Library:CreateWindow(title)
	local ScreenGui = Instance.new("ScreenGui")
	ScreenGui.Parent = LocalPlayer:WaitForChild("PlayerGui")

	-- Frame principal
	local MainFrame = Instance.new("Frame")
	MainFrame.Size = UDim2.new(0, 200, 0, 30) -- altura inicial (só título)
	MainFrame.Position = UDim2.new(0.5, -100, 0.5, -100)
	MainFrame.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
	MainFrame.BorderSizePixel = 0
	MainFrame.Active = true
	MainFrame.Parent = ScreenGui

	-- Barra de título
	local TitleBar = Instance.new("Frame")
	TitleBar.Size = UDim2.new(1, 0, 0, 30)
	TitleBar.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
	TitleBar.BorderSizePixel = 0
	TitleBar.Parent = MainFrame

	local TitleLabel = Instance.new("TextLabel")
	TitleLabel.Size = UDim2.new(1, -30, 1, 0)
	TitleLabel.Position = UDim2.new(0, 5, 0, 0)
	TitleLabel.BackgroundTransparency = 1
	TitleLabel.Text = title
	TitleLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
	TitleLabel.Font = Enum.Font.SourceSansBold
	TitleLabel.TextSize = 14
	TitleLabel.TextXAlignment = Enum.TextXAlignment.Left
	TitleLabel.Parent = TitleBar

	-- Botão de minimizar (-)
	local MinimizeButton = Instance.new("TextButton")
	MinimizeButton.Size = UDim2.new(0, 30, 0, 30)
	MinimizeButton.Position = UDim2.new(1, -30, 0, 0)
	MinimizeButton.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
	MinimizeButton.BorderSizePixel = 0
	MinimizeButton.Text = "-"
	MinimizeButton.TextColor3 = Color3.fromRGB(255, 255, 255)
	MinimizeButton.Font = Enum.Font.SourceSansBold
	MinimizeButton.TextSize = 20
	MinimizeButton.Parent = TitleBar

	-- Container com rolagem para os elementos
	local Container = Instance.new("ScrollingFrame")
	Container.Size = UDim2.new(1, 0, 0, 0) -- altura automática será gerenciada
	Container.Position = UDim2.new(0, 0, 0, 30)
	Container.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
	Container.BorderSizePixel = 0
	Container.ScrollBarThickness = 4
	Container.ScrollBarImageColor3 = Color3.fromRGB(80, 80, 80)
	Container.CanvasSize = UDim2.new(0, 0, 0, 0)
	Container.AutomaticCanvasSize = Enum.AutomaticSize.Y
	Container.Parent = MainFrame

	local UIListLayout = Instance.new("UIListLayout")
	UIListLayout.Padding = UDim.new(0, 2)
	UIListLayout.HorizontalAlignment = Enum.HorizontalAlignment.Center
	UIListLayout.SortOrder = Enum.SortOrder.LayoutOrder
	UIListLayout.Parent = Container

	local Window = {}
	Window.MainFrame = MainFrame
	Window.Container = Container
	Window.Minimized = false

	-- Atualiza altura do frame principal quando o container muda de tamanho
	Container:GetPropertyChangedSignal("AbsoluteSize"):Connect(function()
		if not Window.Minimized then
			MainFrame.Size = UDim2.new(0, 200, 0, 30 + Container.AbsoluteSize.Y)
		end
	end)

	-- Minimizar / expandir
	MinimizeButton.MouseButton1Click:Connect(function()
		Window.Minimized = not Window.Minimized
		Container.Visible = not Window.Minimized
		if Window.Minimized then
			MainFrame.Size = UDim2.new(0, 200, 0, 30)
		else
			MainFrame.Size = UDim2.new(0, 200, 0, 30 + Container.AbsoluteSize.Y)
		end
	end)

	-- Tornar a janela arrastável (PC e toque)
	local dragging = false
	local dragStartPos = nil
	local startFramePos = nil

	TitleBar.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true
			dragStartPos = input.Position
			startFramePos = MainFrame.Position
		end
	end)

	UserInputService.InputChanged:Connect(function(input)
		if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
			local delta = input.Position - dragStartPos
			MainFrame.Position = UDim2.new(
				startFramePos.X.Scale,
				startFramePos.X.Offset + delta.X,
				startFramePos.Y.Scale,
				startFramePos.Y.Offset + delta.Y
			)
		end
	end)

	UserInputService.InputEnded:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = false
		end
	end)

	--- Slider (deslizante)
	function Window:AddSlider(text, min, max, default, callback)
		local frame = Instance.new("Frame")
		frame.Size = UDim2.new(1, -10, 0, 35)
		frame.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
		frame.BorderSizePixel = 0
		frame.Parent = Container

		local label = Instance.new("TextLabel")
		label.Size = UDim2.new(1, 0, 0, 15)
		label.Position = UDim2.new(0, 5, 0, 0)
		label.BackgroundTransparency = 1
		label.Text = text .. ": " .. tostring(default)
		label.TextColor3 = Color3.fromRGB(255, 255, 255)
		label.Font = Enum.Font.SourceSans
		label.TextSize = 12
		label.TextXAlignment = Enum.TextXAlignment.Left
		label.Parent = frame

		local sliderFrame = Instance.new("Frame")
		sliderFrame.Size = UDim2.new(1, -20, 0, 15)
		sliderFrame.Position = UDim2.new(0, 10, 0, 17)
		sliderFrame.BackgroundColor3 = Color3.fromRGB(80, 80, 80)
		sliderFrame.BorderSizePixel = 0
		sliderFrame.Parent = frame

		local fill = Instance.new("Frame")
		fill.Size = UDim2.new(0, 0, 1, 0)
		fill.BackgroundColor3 = Color3.fromRGB(0, 120, 255)
		fill.BorderSizePixel = 0
		fill.Parent = sliderFrame

		local thumb = Instance.new("TextButton")
		thumb.Size = UDim2.new(0, 12, 0, 12)
		thumb.Position = UDim2.new(0, 0, 0.5, -6)
		thumb.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
		thumb.BorderSizePixel = 0
		thumb.Text = ""
		thumb.Parent = sliderFrame

		local value = default
		local function updateVisual(val)
			val = math.clamp(val, min, max)
			local percent = (val - min) / (max - min)
			fill.Size = UDim2.new(percent, 0, 1, 0)
			thumb.Position = UDim2.new(percent, -6, 0.5, -6)
			label.Text = text .. ": " .. string.format("%.2f", val)
		end
		updateVisual(default)

		local function setValueFromX(x)
			local sliderAbsPos = sliderFrame.AbsolutePosition.X
			local sliderWidth = sliderFrame.AbsoluteSize.X
			local relativeX = math.clamp(x - sliderAbsPos, 0, sliderWidth)
			local percent = relativeX / sliderWidth
			value = min + percent * (max - min)
			updateVisual(value)
			callback(value)
		end

		thumb.InputBegan:Connect(function(input)
			if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
				local moveConn, endConn
				moveConn = UserInputService.InputChanged:Connect(function(changedInput)
					if changedInput.UserInputType == Enum.UserInputType.MouseMovement or changedInput.UserInputType == Enum.UserInputType.Touch then
						setValueFromX(changedInput.Position.X)
					end
				end)
				endConn = UserInputService.InputEnded:Connect(function(endedInput)
					if endedInput.UserInputType == Enum.UserInputType.MouseButton1 or endedInput.UserInputType == Enum.UserInputType.Touch then
						moveConn:Disconnect()
						endConn:Disconnect()
					end
				end)
			end
		end)

		return frame
	end

	--- Dropdown (seleciona uma opção)
	function Window:AddDropdown(text, options, default, callback)
		local frame = Instance.new("Frame")
		frame.Size = UDim2.new(1, -10, 0, 25)
		frame.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
		frame.BorderSizePixel = 0
		frame.Parent = Container

		local button = Instance.new("TextButton")
		button.Size = UDim2.new(1, 0, 1, 0)
		button.BackgroundTransparency = 1
		button.Text = text .. ": " .. (default or "Nenhum")
		button.TextColor3 = Color3.fromRGB(255, 255, 255)
		button.Font = Enum.Font.SourceSans
		button.TextSize = 14
		button.TextXAlignment = Enum.TextXAlignment.Left
		button.Parent = frame

		local dropdownFrame = Instance.new("Frame")
		dropdownFrame.Size = UDim2.new(1, 0, 0, 0)
		dropdownFrame.Position = UDim2.new(0, 0, 1, 0)
		dropdownFrame.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
		dropdownFrame.BorderSizePixel = 0
		dropdownFrame.Visible = false
		dropdownFrame.ZIndex = 10
		dropdownFrame.Parent = frame

		local listLayout = Instance.new("UIListLayout")
		listLayout.Parent = dropdownFrame
		listLayout.SortOrder = Enum.SortOrder.LayoutOrder

		local selected = default
		local function rebuildOptions()
			for _, child in ipairs(dropdownFrame:GetChildren()) do
				if child:IsA("TextButton") then child:Destroy() end
			end
			for i, option in ipairs(options) do
				local optBtn = Instance.new("TextButton")
				optBtn.Size = UDim2.new(1, 0, 0, 20)
				optBtn.BackgroundColor3 = (option == selected) and Color3.fromRGB(60, 60, 60) or Color3.fromRGB(50, 50, 50)
				optBtn.BorderSizePixel = 0
				optBtn.Text = option
				optBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
				optBtn.Font = Enum.Font.SourceSans
				optBtn.TextSize = 14
				optBtn.Parent = dropdownFrame
				optBtn.LayoutOrder = i
				optBtn.MouseButton1Click:Connect(function()
					selected = option
					button.Text = text .. ": " .. selected
					dropdownFrame.Visible = false
					frame.Size = UDim2.new(1, -10, 0, 25)
					callback(selected)
				end)
			end
			dropdownFrame.Size = UDim2.new(1, 0, 0, 20 * #options + 2)
		end
		rebuildOptions()

		button.MouseButton1Click:Connect(function()
			dropdownFrame.Visible = not dropdownFrame.Visible
			if dropdownFrame.Visible then
				frame.Size = UDim2.new(1, -10, 0, 25 + dropdownFrame.AbsoluteSize.Y)
			else
				frame.Size = UDim2.new(1, -10, 0, 25)
			end
		end)

		return frame
	end

	--- MultiDropdown (seleciona várias opções)
	function Window:AddMultiDropdown(text, options, defaults, callback)
		defaults = defaults or {}
		local selected = {}
		for _, opt in ipairs(options) do
			selected[opt] = defaults[opt] or false
		end

		local frame = Instance.new("Frame")
		frame.Size = UDim2.new(1, -10, 0, 25)
		frame.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
		frame.BorderSizePixel = 0
		frame.Parent = Container

		local button = Instance.new("TextButton")
		button.Size = UDim2.new(1, 0, 1, 0)
		button.BackgroundTransparency = 1
		local function updateText()
			local sel = {}
			for opt, isSel in pairs(selected) do
				if isSel then table.insert(sel, opt) end
			end
			button.Text = text .. ": " .. (#sel > 0 and table.concat(sel, ", ") or "Nenhum")
		end
		updateText()
		button.TextColor3 = Color3.fromRGB(255, 255, 255)
		button.Font = Enum.Font.SourceSans
		button.TextSize = 14
		button.TextXAlignment = Enum.TextXAlignment.Left
		button.Parent = frame

		local dropdownFrame = Instance.new("Frame")
		dropdownFrame.Size = UDim2.new(1, 0, 0, 0)
		dropdownFrame.Position = UDim2.new(0, 0, 1, 0)
		dropdownFrame.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
		dropdownFrame.BorderSizePixel = 0
		dropdownFrame.Visible = false
		dropdownFrame.ZIndex = 10
		dropdownFrame.Parent = frame

		local listLayout = Instance.new("UIListLayout")
		listLayout.Parent = dropdownFrame
		listLayout.SortOrder = Enum.SortOrder.LayoutOrder

		local function rebuildOptions()
			for _, child in ipairs(dropdownFrame:GetChildren()) do
				if child:IsA("Frame") then child:Destroy() end
			end
			for i, option in ipairs(options) do
				local optFrame = Instance.new("Frame")
				optFrame.Size = UDim2.new(1, 0, 0, 20)
				optFrame.BackgroundColor3 = selected[option] and Color3.fromRGB(60, 60, 60) or Color3.fromRGB(50, 50, 50)
				optFrame.BorderSizePixel = 0
				optFrame.Parent = dropdownFrame
				optFrame.LayoutOrder = i

				local checkLabel = Instance.new("TextLabel")
				checkLabel.Size = UDim2.new(0, 15, 1, 0)
				checkLabel.Position = UDim2.new(0, 2, 0, 0)
				checkLabel.BackgroundTransparency = 1
				checkLabel.Text = selected[option] and "✓" or ""
				checkLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
				checkLabel.Font = Enum.Font.SourceSans
				checkLabel.TextSize = 14
				checkLabel.Parent = optFrame

				local optLabel = Instance.new("TextLabel")
				optLabel.Size = UDim2.new(1, -20, 1, 0)
				optLabel.Position = UDim2.new(0, 20, 0, 0)
				optLabel.BackgroundTransparency = 1
				optLabel.Text = option
				optLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
				optLabel.Font = Enum.Font.SourceSans
				optLabel.TextSize = 14
				optLabel.TextXAlignment = Enum.TextXAlignment.Left
				optLabel.Parent = optFrame

				optFrame.InputBegan:Connect(function(input)
					if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
						selected[option] = not selected[option]
						checkLabel.Text = selected[option] and "✓" or ""
						optFrame.BackgroundColor3 = selected[option] and Color3.fromRGB(60, 60, 60) or Color3.fromRGB(50, 50, 50)
						updateText()
						callback(option, selected[option], selected)
					end
				end)
			end
			dropdownFrame.Size = UDim2.new(1, 0, 0, 20 * #options + 2)
		end
		rebuildOptions()

		button.MouseButton1Click:Connect(function()
			dropdownFrame.Visible = not dropdownFrame.Visible
			if dropdownFrame.Visible then
				frame.Size = UDim2.new(1, -10, 0, 25 + dropdownFrame.AbsoluteSize.Y)
			else
				frame.Size = UDim2.new(1, -10, 0, 25)
			end
		end)

		return frame
	end

	--- Keybind (captura uma tecla)
	function Window:AddKeybind(text, defaultKeyCode, callback)
		local keyCode = defaultKeyCode or Enum.KeyCode.Unknown
		local frame = Instance.new("Frame")
		frame.Size = UDim2.new(1, -10, 0, 25)
		frame.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
		frame.BorderSizePixel = 0
		frame.Parent = Container

		local label = Instance.new("TextLabel")
		label.Size = UDim2.new(1, -70, 1, 0)
		label.Position = UDim2.new(0, 5, 0, 0)
		label.BackgroundTransparency = 1
		label.Text = text
		label.TextColor3 = Color3.fromRGB(255, 255, 255)
		label.Font = Enum.Font.SourceSans
		label.TextSize = 14
		label.TextXAlignment = Enum.TextXAlignment.Left
		label.Parent = frame

		local keyButton = Instance.new("TextButton")
		keyButton.Size = UDim2.new(0, 60, 0, 25)
		keyButton.Position = UDim2.new(1, -65, 0, 0)
		keyButton.BackgroundColor3 = Color3.fromRGB(70, 70, 70)
		keyButton.BorderSizePixel = 0
		keyButton.Text = keyCode.Name
		keyButton.TextColor3 = Color3.fromRGB(255, 255, 255)
		keyButton.Font = Enum.Font.SourceSans
		keyButton.TextSize = 12
		keyButton.Parent = frame

		local listening = false
		local inputConn

		keyButton.MouseButton1Click:Connect(function()
			if listening then
				listening = false
				keyButton.Text = keyCode.Name
				if inputConn then inputConn:Disconnect() end
				return
			end
			listening = true
			keyButton.Text = "..."
			inputConn = UserInputService.InputBegan:Connect(function(input, gameProcessed)
				if not gameProcessed and input.UserInputType == Enum.UserInputType.Keyboard then
					keyCode = input.KeyCode
					listening = false
					keyButton.Text = keyCode.Name
					inputConn:Disconnect()
					callback(keyCode)
				end
			end)
		end)

		return frame
	end

		-- Sliders RGB
		local function makeColorSlider(name, yPos, initial)
			local sliderFrame = Instance.new("Frame")
			sliderFrame.Size = UDim2.new(1, -10, 0, 25)
			sliderFrame.Position = UDim2.new(0, 5, 0, yPos)
			sliderFrame.BackgroundTransparency = 1
			sliderFrame.Parent = pickerFrame

			local lbl = Instance.new("TextLabel")
			lbl.Size = UDim2.new(0, 15, 1, 0)
			lbl.BackgroundTransparency = 1
			lbl.Text = name
			lbl.TextColor3 = Color3.fromRGB(255, 255, 255)
			lbl.Font = Enum.Font.SourceSansBold
			lbl.TextSize = 12
			lbl.Parent = sliderFrame

			local bg = Instance.new("Frame")
			bg.Size = UDim2.new(1, -25, 1, 0)
			bg.Position = UDim2.new(0, 20, 0, 0)
			bg.BackgroundColor3 = Color3.fromRGB(60, 60, 60)

			thumb.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
			thumb.BorderSizePixel = 0

			thumb.Parent = bg

			local val = initial
			local function updateVisual(v)
				v = math.clamp(v, 0, 255)
				local percent = v / 255
				fill.Size = UDim2.new(percent, 0, 1, 0)
				thumb.Position = UDim2.new(percent, -5, 0, 0)

			end
			updateVisual(val)

			local function setFromX(x)
				local absX = bg.AbsolutePosition.X
				local width = bg.AbsoluteSize.X
				local rel = math.clamp(x - absX, 0, width)
				val = math.floor((rel / width) * 255 + 0.5)
				updateVisual(val)
				return val
			end

			thumb.InputBegan:Connect(function(input)
				if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
					local moveConn, endConn
					moveConn = UserInputService.InputChanged:Connect(function(changedInput)
						if changedInput.UserInputType == Enum.UserInputType.MouseMovement or changedInput.UserInputType == Enum.UserInputType.Touch then
							val = setFromX(changedInput.Position.X)
							updateColor()
						end
					end)
					endConn = UserInputService.InputEnded:Connect(function(endedInput)
						if endedInput.UserInputType == Enum.UserInputType.MouseButton1 or endedInput.UserInputType == Enum.UserInputType.Touch then
							moveConn:Disconnect()
							endConn:Disconnect()
						end
					end)
				end
			end)

			return {
				SetValue = function(v) val = updateVisual(v) end,
				GetValue = function() return val end
			}
		end

		local rSlider = makeColorSlider("R", 5, math.floor(currentColor.R * 255 + 0.5))
		local gSlider = makeColorSlider("G", 35, math.floor(currentColor.G * 255 + 0.5))
		local bSlider = makeColorSlider("B", 65, math.floor(currentColor.B * 255 + 0.5))

		local preview = Instance.new("Frame")
		preview.Size = UDim2.new(1, -10, 0, 20)
		preview.Position = UDim2.new(0, 5, 0, 100)
		preview.BackgroundColor3 = currentColor
		preview.BorderSizePixel = 0
		preview.Parent = pickerFrame

		local function updateColor()
			local r = rSlider.GetValue()
			local g = gSlider.GetValue()
			local b = bSlider.GetValue()
			currentColor = Color3.fromRGB(r, g, b)
			colorButton.BackgroundColor3 = currentColor
			preview.BackgroundColor3 = currentColor
		end

		colorButton.MouseButton1Click:Connect(function()
			pickerFrame.Visible = not pickerFrame.Visible
			if pickerFrame.Visible then
				frame.Size = UDim2.new(1, -10, 0, 25 + pickerFrame.AbsoluteSize.Y + 5)
			else
				frame.Size = UDim2.new(1, -10, 0, 25)
				callback(currentColor)
			end
		end)

		return frame
	end

	-- Retorna a janela criada
	return Window
end

return Library
